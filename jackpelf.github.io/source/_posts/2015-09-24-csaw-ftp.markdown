	__int64 __fastcall calc(__int64 a1)
	{
	  int i; // [sp+10h] [bp-8h]@1
	  int resum; // [sp+14h] [bp-4h]@1

	  resum = 5381;
	  for ( i = 0; *(_BYTE *)(i + a1); ++i )
	    resum = 33 * resum + *(_BYTE *)(i + a1);
	  return (unsigned int)resum;
	}
