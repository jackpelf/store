###0x00 
fork 在父进程中返回子进程的pid，在子进程中返回0

###0x01

	#include<stdio.h>
	#include<unistd.h>
	int main()
	{
		pid_t 	fpid;
		fpid = fork();
		if(fpid == 0)
		{
			puts("i am child");
			printf("my pid is %d\n", getpid());
		}
		else
		{
			puts("i am parents");
			printf("my pid is %d\n", getpid());
		}
	}
