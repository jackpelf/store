        private void button1_Click(object sender, EventArgs e)
        {
            char[] arr = textBox1.Text.ToCharArray();
            int i;
            for (i = 0; i < arr.Length; i++)
            {
                int tmp = (int)arr[i];
                if (tmp >= 97 && tmp <= (int)'z')
                    tmp = 97 + (tmp - 97 + 5) % 26;
                else if (tmp >= 65 && tmp <= 'Z')
                    tmp = 65 + (tmp - 65 + 5) % 26;
                arr[i] = (char)tmp;
            }
            textBox2.Text = new string(arr);
        }
