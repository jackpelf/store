#import <Foundation/Foundation.h>
NSString* Cipher(NSString *text)
{
	NSLog(@"length is %d",[text length]);
	int i;
	char buf[2000];
	NSLog(@"Answer is: "); 
	for(i = 0; i < [text length]; i++)
	{
		int tmp = [text characterAtIndex:i];
		if( tmp >= 'a' && tmp <= 'z')		
			buf[i] = 'a' + (tmp - 'a' + 5) % 26 ;
		else if( tmp >= 'A' && tmp <= 'Z')		
			buf[i] = 'A' + (tmp - 'A' + 5) % 26 ;
		else 
			buf[i] = tmp;
	}
	NSString *ans= [[NSString alloc] initWithCString:buf];
	return ans;
}
