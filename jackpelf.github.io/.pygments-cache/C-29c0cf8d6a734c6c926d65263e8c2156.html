int __cdecl reorder(int a1)
{
  int result; // eax@1
  int v2; // esi@2
  int t; // eax@2
  int v4; // eax@4

  result = a1;
  do
  {
    v2 = 3 * result;
    t = table1[3 * result];
    if ( t != -1 )                              // -1是一个标识
      reorder(t);
    v4 = i;
    hex1[i] = cphex1[v2 * 4];                   // table3选择一个元素乘以4作为要置换的index，
                                                // 所以table3是置换表
                                                // 选择的序列是4倍的关系
    i = v4 + 1;
    result = table3[v2];
  }
  while ( result != -1 );                       // table1选择一个元素作为是否结束递归的条件
  return result;
}
